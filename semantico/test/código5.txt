{

    float arg4, arg5, arg6; //declaracao de floats
    arg1 = 0.5;
    arg2 = 1.75;
    arg4 = 5; 
    arg5 = 7;
    
    arg6 = (int) arg1 + (float) arg2 * arg4; //realizacao uma operação
    arg3 = arg2 - arg1 * arg6; //realiza outra operacao

    printf ("%i\n",arg3);
